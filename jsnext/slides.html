Block-scoped vars (continued)

ES6

New keyword: let 

console.log(i); //ReferenceError: i is not defined
if (false) {
  let i = 10;
}

if (false) {
  let i = 10;
}
console.log(i); //ReferenceError: i is not defined

for (let i = 0; i < 10; i++) {
  console.log(i);
}
console.log(i); //ReferenceError: i is not defined

Arrow functions

ES6

users.forEach(function (user, index) {
  this.processUser(user);
  //TypeError: Object window no method 'processUser'
});

users.forEach((user, index) => {
  this.processUser(user);
  // arrow functions have the scope you actually want.
});

Classes

ES6

class Polygon {
  constructor(height, width) { //class constructor
    this.name = 'Polygon';
    this.height = height;
    this.width = width;
  }

  sayName() { //class method
    console.log('Hi, I am a', this.name + '.');
  }
}

Classes (continued)

ES6

class Square extends Polygon {
  constructor(length) {
    super(length, length); //call the parent method with super
    this.name = 'Square';
  }
  get area() { //calculated attribute getter
    return this.height * this.width;
  }
}
let s = new Square(5);
console.log(s.area);

$http
