Generators (yield)

ES6

function *range(begin, end) {
  for (let i = begin; i < end; ++i) {
    yield i;
  }
}

for (let i of range(0, 10)) {
  console.log(i); // 0, 1, 2, ... 10
}

let generator = range(0, 10);
console.log(generator.next()) // 0
console.log(generator.next()) // 1

Generators (yield)

Classes

ES6

class Polygon {
  constructor(height, width) { //class constructor
    this.name = 'Polygon';
    this.height = height;
    this.width = width;
  }

  sayName() { //class method
    console.log('Hi, I am a', this.name + '.');
  }
}

Classes (continued)

ES6

class Square extends Polygon {
  constructor(length) {
    super(length, length); //call the parent method with super
    this.name = 'Square';
  }
  get area() { //calculated attribute getter
    return this.height * this.width;
  }
}
let s = new Square(5);
console.log(s.area);

Basic Concepts

$http

  • The most basic way of accessing remote resourses
  • Has convenience methods for performing all the basic rest methods GET, POST, DELETE, and PUT.
  • Returns a promise, but also has convenience methods for success and error
  $http.get("/some/url/items.json")
    .then(function (data) {
      $scope.items = data;
    }, function() {
      alert("Error occurred");
    });
